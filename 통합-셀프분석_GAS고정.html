/* ═══════════════════════════════════════════════════════════ DRAFT (임시저장 파일 / 불러오기 / 초기화) ═══════════════════════════════════════════════════════════ */ function makeSnapshot(isDraft){ saveCurrentPassage(); saveAllExplanations(); const name = $('s-name')?.value?.trim() || ''; return { assignment: ASGN, studentName: name, passages: DATA, explanations: EXPL, modes: MODES, qsts: QSTS, cc: CC, wc: WC, savedAt: new Date().toISOString(), isDraft: !!isDraft }; } function applySnapshot(snap){ // ASGN은 loadStudent에서 초기화되므로, loadStudent 먼저 실행 후 덮어씌웁니다. if(snap.studentName) $('s-name').value = snap.studentName; if(snap.passages) DATA = snap.passages; if(snap.explanations) EXPL = snap.explanations; if(snap.modes) MODES = snap.modes; if(snap.qsts) QSTS = snap.qsts; if(snap.cc) CC = snap.cc; if(snap.wc) WC = snap.wc; // 로컬 자동저장도 즉시 반영 autoSave(); renderPassage(CUR); updateTabs(); updateProgress(); } function saveDraftFile(){ if(!ASGN){ toast('⚠️ 과제를 불러온 뒤 사용하세요'); return; } const name = $('s-name')?.value?.trim() || ''; if(!name){ toast('⚠️ 이름을 입력해주세요'); return; } const snap = makeSnapshot(true); const safeTitle = (ASGN.title||'과제').replace(/[\/\\:*?"<>|]/g,'_'); const filename = `임시저장_${name}_${safeTitle}.html`; downloadBakedFile(snap,'draft',filename); toast('✅ 임시저장 파일 저장됨! (다음에 이 파일로 이어쓰기)', 3200); } function openDraftImport(){ const inp = $('draft-file'); if(!inp) return; inp.value = ''; inp.onchange = async (e)=>{ const f = e.target.files && e.target.files[0]; if(!f) return; const text = await f.text(); // 1) JSON 단독 파일인 경우 try{ const obj = JSON.parse(text); if(obj && obj.assignment && obj.passages){ loadDraft(obj); return; } }catch(_){} // 2) HTML baked 파일인 경우: window.__BAKED_DRAFT__를 파싱하기 어려우므로, // 사용자가 '열기'로 실행하면 자동 로딩되도록 설계됨. // 여기서는 안내만 합니다. toast('📌 임시저장 HTML은 "파일 열기"로 실행하면 자동으로 이어서 로딩됩니다.', 4200); }; inp.click(); } function loadDraft(d){ if(!d?.assignment?.passages?.length){ toast('⚠️ 임시저장 파일이 올바르지 않습니다'); return; } // 과제 로딩 loadStudent(d.assignment); // 스냅샷 적용 setTimeout(()=>applySnapshot(d), 0); toast('✅ 임시저장 내용 불러오기 완료', 2600); } function clearLocalWork(){ if(!ASGN) return; const key = 'sna_' + ASGN.title; localStorage.removeItem(key); toast('🧹 이 과제의 자동저장 기록을 삭제했습니다.', 2600); } /* ═══════════════════════════════════════════════════════════ PDF Export (다운로드 + 모바일이면 공유 시도) ═══════════════════════════════════════════════════════════ */ async function exportPdf(){ const view = document.querySelector('#view-review:not(.hide)') ? 'review' : (document.querySelector('#view-student:not(.hide)') ? 'student' : 'gen'); if(view==='gen'){ toast('⚠️ 학생용/검토 화면에서 PDF 저장이 가능합니다'); return; } // 학생용: 최신 입력 저장 if(view==='student'){ saveCurrentPassage(); saveAllExplanations(); } const el = (view==='review') ? $('view-review') : $('view-student'); const aTitle = (view==='review') ? (RDATA?.assignment?.title||'과제') : (ASGN?.title||'과제'); const sName = (view==='review') ? (RDATA?.studentName||'학생') : ($('s-name')?.value?.trim()||'학생'); const safeTitle = aTitle.replace(/[\/\\:*?"<>|]/g,'_'); const filename = `${sName}_${safeTitle}.pdf`; toast('⏳ PDF 생성 중...', 2000); const opt = { margin: 6, filename: filename, image: { type: 'jpeg', quality: 0.95 }, html2canvas: { scale: 2, useCORS: true, backgroundColor: '#ffffff' }, jsPDF: { unit: 'mm', format: 'a4', orientation: 'portrait' } }; // pdf 저장(다운로드) try{ await html2pdf().set(opt).from(el).save(); toast('✅ PDF 저장 완료! 카톡에서 "파일"로 전송하세요.', 3500); }catch(e){ console.error(e); toast('⚠️ PDF 생성에 실패했습니다. "인쇄 → PDF 저장"도 가능합니다.', 3800); } }
